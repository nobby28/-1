// ─── Helpers ─────────────────────────────────────────────────────────────────
function getDeliveryColIndex(month) {
  const idx = MONTH_ORDER.indexOf(month);
  return idx === -1 ? -1 : COL_FIRST_MONTH_DELIVERY + idx * 4;
}

function normalizeJAN(jan) {
  if (!jan) return "";
  return String(jan)
    .replace(/[Ａ-Ｚａ-ｚ０-９]/g, c => String.fromCharCode(c.charCodeAt(0) - 0xFEE0))
    .replace(/[\s\-.]/g, "").trim();
}

function fileToBase64(file) {
  return new Promise((res, rej) => {
    const r = new FileReader();
    r.onload = () => res(r.result.split(",")[1]);
    r.onerror = rej;
    r.readAsDataURL(file);
  });
}

function todayMonth() { return new Date().getMonth() + 1; }

// ─── Pure-JS XLSX reader (no external dependency) ────────────────────────────
function u16(b, o) { return b[o] | (b[o+1]<<8); }
function u32(b, o) { return (b[o]|(b[o+1]<<8)|(b[o+2]<<16)|(b[o+3]<<24))>>>0; }
const td = new TextDecoder("utf-8");

async function inflateRaw(data) {
  const ds = new DecompressionStream("deflate-raw");
  const w = ds.writable.getWriter();
  const r = ds.readable.getReader();
  w.write(data); w.close();
  const chunks = [];
  for (;;) { const {done,value} = await r.read(); if(done) break; chunks.push(value); }
  const total = chunks.reduce((s,c)=>s+c.length,0);
  const out = new Uint8Array(total); let off=0;
  for (const c of chunks) { out.set(c,off); off+=c.length; }
  return out;
}

async function readXlsx(buf) {
  const b = new Uint8Array(buf);
  // find EOCD
  let eocd=-1;
  for (let i=b.length-22;i>=0;i--) {
    if(b[i]===0x50&&b[i+1]===0x4B&&b[i+2]===0x05&&b[i+3]===0x06){eocd=i;break;}
  }
  if(eocd<0) throw new Error("Invalid XLSX");
  const cdOff=u32(b,eocd+16), cdSz=u32(b,eocd+12);
  const entries={};
  let pos=cdOff;
  while(pos<cdOff+cdSz){
    if(!(b[pos]===0x50&&b[pos+1]===0x4B&&b[pos+2]===0x01&&b[pos+3]===0x02)) break;
    const fnLen=u16(b,pos+28),exLen=u16(b,pos+30),cmLen=u16(b,pos+32),lhOff=u32(b,pos+42);
    const fname=td.decode(b.slice(pos+46,pos+46+fnLen));
    entries[fname]=lhOff;
    pos+=46+fnLen+exLen+cmLen;
  }

  async function read(name) {
    const lhOff=entries[name]; if(lhOff==null) return null;
    const fnLen=u16(b,lhOff+26),exLen=u16(b,lhOff+28),method=u16(b,lhOff+8);
    const compSz=u32(b,lhOff+18),dataStart=lhOff+30+fnLen+exLen;
    const comp=b.slice(dataStart,dataStart+compSz);
    if(method===0) return td.decode(comp);
    if(method===8) return td.decode(await inflateRaw(comp));
    return null;
  }

  // Parse shared strings
  const ssXml = await read("xl/sharedStrings.xml");
  const ss=[];
  if(ssXml){
    for(const m of ssXml.matchAll(/<si>([\s\S]*?)<\/si>/g)){
      const texts=[...m[1].matchAll(/<t(?:[^>]*)>([\s\S]*?)<\/t>/g)].map(t=>
        t[1].replace(/&amp;/g,"&").replace(/&lt;/g,"<").replace(/&gt;/g,">")
            .replace(/&quot;/g,'"').replace(/&#(\d+);/g,(_,n)=>String.fromCharCode(+n))
      );
      ss.push(texts.join(""));
    }
  }

  // Find sheet path by name
  const wbXml=await read("xl/workbook.xml");
  const wbRels=await read("xl/_rels/workbook.xml.rels");
  let sheetPath=null;
  if(wbXml&&wbRels){
    for(const sm of (wbXml.match(/<sheet[^>]+>/g)||[])){
      const nameM=sm.match(/name="([^"]*)"/), ridM=sm.match(/r:id="([^"]*)"/);
      if(nameM&&ridM&&nameM[1]===SHEET_NAME){
        const rm=wbRels.match(new RegExp(`Id="${ridM[1]}"[^>]+Target="([^"]+)"`));
        if(rm) sheetPath="xl/"+rm[1].replace(/^\.\//,"");
      }
    }
  }
  if(!sheetPath){
    for(const c of ["xl/worksheets/sheet2.xml","xl/worksheets/sheet1.xml"]){
      if(entries[c]!=null){sheetPath=c;break;}
    }
  }
  if(!sheetPath) throw new Error(`シート「${SHEET_NAME}」が見つかりません`);

  const shXml=await read(sheetPath);
  if(!shXml) throw new Error("シートXMLが読み込めません");

  // Parse rows → 2D array
  const rows=[];
  for(const rm of shXml.matchAll(/<row[^>]*r="(\d+)"[^>]*>([\s\S]*?)<\/row>/g)){
    const ri=parseInt(rm[1])-1;
    const rowXml=rm[2];
    const rowArr=[];
    for(const cm of rowXml.matchAll(/<c r="([A-Z]+)(\d+)"([^>]*)>([\s\S]*?)<\/c>/g)){
      const colStr=cm[1],attrs=cm[3],inner=cm[4];
      const colIdx=colStr.split("").reduce((acc,ch)=>acc*26+(ch.charCodeAt(0)-64),0)-1;
      const vM=inner.match(/<v>([\s\S]*?)<\/v>/);
      if(!vM) continue;
      let val=vM[1];
      if(attrs.includes('t="s"')) val=ss[parseInt(val)]??val;
      else { const n=parseFloat(val); if(!isNaN(n)) val=n; }
      while(rowArr.length<=colIdx) rowArr.push(null);
      rowArr[colIdx]=val;
    }
    while(rows.length<=ri) rows.push(null);
    rows[ri]=rowArr;
  }
  return rows;
}

function buildMap(rows) {
  const map=new Map();
  for(let i=DATA_START_ROW;i<rows.length;i++){
    const row=rows[i]; if(!row||!row[COL_JAN]) continue;
    const jan=normalizeJAN(row[COL_JAN]); if(jan.length<8) continue;
    const deliveries={};
    MONTH_ORDER.forEach(m=>{ deliveries[m]=row[getDeliveryColIndex(m)]??null; });
    map.set(jan,{rowIdx:i, name:String(row[3]||"").trim(), manufacturer:String(row[2]||"").trim(), deliveries});
  }
  return map;
}

// ─── CSV export ───────────────────────────────────────────────────────────────
function exportCSV(inventoryMap, fname) {
  const hdr=["JANコード","品名","メーカー",
    ...MONTH_ORDER.flatMap(m=>[`${m}月納品ケース数`])
  ];
  const rows=[...inventoryMap.entries()].map(([jan,v])=>[
    jan, v.name, v.manufacturer,
    ...MONTH_ORDER.map(m=>v.deliveries[m]??""   )
  ]);
  const csv=[hdr,...rows].map(r=>r.map(x=>`"${String(x).replace(/"/g,'""')}"`).join(",")).join("\n");
  const blob=new Blob(["\uFEFF"+csv],{type:"text/csv;charset=utf-8;"});
  Object.assign(document.createElement("a"),{href:URL.createObjectURL(blob),download:fname}).click();
}

async function callAI(base64,mediaType){
  const sys=`あなたは日本語の「納品仕切書」から情報を正確に抽出する専門家です。
以下のJSON形式のみで返答してください（説明・マークダウン記号は一切不要）：
{"invoice_number":"伝票番号","invoice_date":"YYYYMMDD","supplier":"仕入先","items":[{"jan_code":"13桁JANコード","product_name":"品名","quantity":数値,"unit_price":数値,"amount":数値}],"total_amount":数値}
JANコードは13桁数字のみ。読取不能はnull。`;
  const res=await fetch("https://api.anthropic.com/v1/messages",{
    method:"POST",headers:{"Content-Type":"application/json"},
    body:JSON.stringify({model:"claude-sonnet-4-20250514",max_tokens:2000,system:sys,
      messages:[{role:"user",content:[
        {type:"image",source:{type:"base64",media_type:mediaType,data:base64}},
        {type:"text",text:"この納品仕切書をJSONで出力してください。"}
      ]}]})
  });
  const d=await res.json();
  const t=(d.content||[]).map(b=>b.text||"").join("").replace(/```json|```/g,"").trim();
  return JSON.parse(t);
}

// ─── DropZone ─────────────────────────────────────────────────────────────────
function DropZone({onFile,accept,label,sub,icon,color=T.teal}){
  const [over,setOver]=useState(false);
  const ref=useRef();
  return(
    <div onDragOver={e=>{e.preventDefault();setOver(true);}} onDragLeave={()=>setOver(false)}
      onDrop={e=>{e.preventDefault();setOver(false);const f=e.dataTransfer.files[0];if(f)onFile(f);}}
      onClick={()=>ref.current?.click()}
      style={{border:`2px dashed ${over?color:T.border}`,borderRadius:12,padding:"18px",textAlign:"center",cursor:"pointer",background:over?color+"0A":T.surfaceAlt,transition:"all 0.2s"}}>

      <div style={{fontSize:12,color:T.textLight,marginTop:2}}>{sub}</div>
      <input ref={ref} type="file" accept={accept} onChange={e=>{if(e.target.files[0])onFile(e.target.files[0]);}} style={{display:"none"}}/>
    </div>
  );
}

// ─── Scanner ─────────────────────────────────────────────────────────────────
function Scanner({onResult}){
  const [mode,setMode]=useState("idle");
  const [src,setSrc]=useState(null);
  const [file,setFile]=useState(null);
  const [err,setErr]=useState(null);
  const vidRef=useRef(); const streamRef=useRef(); const fileRef=useRef();
  const stop=useCallback(()=>{streamRef.current?.getTracks().forEach(t=>t.stop());streamRef.current=null;},[]);
  useEffect(()=>()=>stop(),[stop]);

  const startCam=async()=>{
    setErr(null);
    try{
      const s=await navigator.mediaDevices.getUserMedia({video:{facingMode:"environment",width:{ideal:1920}}});
      streamRef.current=s; setMode("camera");
      setTimeout(()=>{if(vidRef.current)vidRef.current.srcObject=s;},80);
    }catch{setErr("カメラアクセスが拒否されました。画像ファイルを選択してください。");}
  };
  const snap=useCallback(()=>{
    if(!vidRef.current)return;
    const c=document.createElement("canvas");
    c.width=vidRef.current.videoWidth;c.height=vidRef.current.videoHeight;
    c.getContext("2d").drawImage(vidRef.current,0,0);
    c.toBlob(blob=>{const f=new File([blob],"snap.jpg",{type:"image/jpeg"});setFile(f);setSrc(c.toDataURL("image/jpeg"));stop();setMode("preview");},"image/jpeg",0.95);
  },[stop]);

  const pickFile=f=>{setFile(f);setSrc(URL.createObjectURL(f));setMode("preview");setErr(null);};
  const analyze=async()=>{
    if(!file)return; setMode("analyzing"); setErr(null);
    try{
      const b64=await fileToBase64(file);
      const result=await callAI(b64,file.type||"image/jpeg");
      onResult(result); reset();
    }catch(e){setErr("読み取り失敗: "+(e.message||"不明"));setMode("preview");}
  };
  const reset=()=>{stop();setMode("idle");setSrc(null);setFile(null);setErr(null);if(fileRef.current)fileRef.current.value="";};

  return(
    <Card style={{padding:22,marginBottom:20}}>
      <div style={{display:"flex",alignItems:"center",gap:12,marginBottom:16}}>
        <div style={{width:42,height:42,borderRadius:11,background:T.navy,display:"flex",alignItems:"center",justifyContent:"center",fontSize:20,flexShrink:0}}>📷</div>
        <div>
          <div style={{fontWeight:800,fontSize:15,color:T.navy}}>納品仕切書スキャン</div>
          <div style={{fontSize:12,color:T.textMid}}>撮影または画像選択 → AIがJANコード・数量を自動抽出</div>
        </div>
      </div>

      {err&&<div style={{background:T.redBg,border:`1px solid ${T.red}40`,borderRadius:8,padding:"8px 14px",marginBottom:12,fontSize:12,color:T.red}}>⚠️ {err}</div>}

      {mode==="idle"&&(
        <div>
          <div style={{display:"grid",gridTemplateColumns:"1fr 1fr",gap:10,marginBottom:12}}>
            <Btn onClick={startCam} bg={T.navy} size="lg">📷 カメラ撮影</Btn>
            <Btn onClick={()=>fileRef.current?.click()} bg={T.navyMid} size="lg">📁 画像を選択</Btn>
          </div>
          <input ref={fileRef} type="file" accept="image/*" onChange={e=>{if(e.target.files[0])pickFile(e.target.files[0]);}} style={{display:"none"}}/>
          <div style={{padding:"12px",border:`2px dashed ${T.border}`,borderRadius:10,color:T.textLight,fontSize:12,background:T.surfaceAlt,textAlign:"center"}}>
            💡 納品仕切書を撮影 → AIがJANコードと数量を読み取り → 選択した月の「納品ケース数」に反映
          </div>
        </div>
      )}

      {mode==="camera"&&(
        <>
          <div style={{position:"relative",borderRadius:12,overflow:"hidden",background:"#000",marginBottom:12}}>

            </div>

            <Btn onClick={snap} bg={T.teal}>📸 撮影する</Btn>
            <Btn onClick={reset} bg={T.textMid} size="sm">キャンセル</Btn>
          </div>
        </>
      )}

      {mode==="preview"&&(
        <>
          <div style={{borderRadius:12,overflow:"hidden",marginBottom:12,border:`2px solid ${T.teal}`}}>
            <img src={src} alt="preview" style={{width:"100%",maxHeight:340,objectFit:"contain",display:"block",background:T.surfaceAlt}}/>

        </table>
      </div>

      {invoice.total_amount&&<div style={{textAlign:"right",color:T.navy,fontWeight:800,fontSize:14,marginBottom:12}}>合計: ¥{Number(invoice.total_amount).toLocaleString()}</div>}

      <div style={{display:"flex",gap:10,justifyContent:"flex-end",flexWrap:"wrap"}}>
        <Btn onClick={onDiscard} bg={T.textMid} size="sm">✕ 破棄</Btn>
        <Btn onClick={()=>onApply(checked,month,writeMode)} disabled={checked.length===0}>
          ✓ {checked.length}件を {month}月に反映
        </Btn>
      </div>
    </Card>
  );
}

// ─── History ──────────────────────────────────────────────────────────────────
function HistoryPanel({history}){
  if(!history.length) return null;
  return(
    <Card style={{padding:18,marginBottom:20}}>
      <div style={{fontWeight:800,fontSize:14,color:T.navy,marginBottom:10}}>🕘 反映履歴（今セッション）</div>
      {history.slice().reverse().map((h,i)=>(
        <div key={i} style={{display:"flex",gap:8,alignItems:"center",padding:"6px 10px",borderRadius:8,background:T.surfaceAlt,marginBottom:4,flexWrap:"wrap"}}>
          <Chip color={T.teal}>{h.month}月</Chip>
          <span style={{fontWeight:700,color:T.navy,fontSize:13}}>{h.name}</span>
          <span style={{fontFamily:"monospace",color:T.textLight,fontSize:11}}>{h.jan}</span>
          <span style={{marginLeft:"auto",fontWeight:800,color:T.tealDark,fontSize:14}}>{h.newVal} ケース</span>
          {h.previous!=null&&<span style={{color:T.textLight,fontSize:12}}>（前: {h.previous}）</span>}
        </div>
      ))}
    </Card>
  );
}

// ─── Inventory Table ──────────────────────────────────────────────────────────
function InventoryTable({inventoryMap,highlightJans,currentMonth}){
  const [search,setSearch]=useState("");
  const [filterMonth,setFilterMonth]=useState(null);
  const [showAll,setShowAll]=useState(false);

  const entries=useMemo(()=>{
    let arr=[...inventoryMap.entries()];
    if(search){const q=search.toLowerCase();arr=arr.filter(([jan,v])=>v.name.toLowerCase().includes(q)||jan.includes(q)||v.manufacturer.toLowerCase().includes(q));}
    if(filterMonth) arr=arr.filter(([,v])=>Number(v.deliveries[filterMonth])>0);
    return arr;
  },[inventoryMap,search,filterMonth]);

  const visible=showAll?entries:entries.slice(0,40);

  return(
    <Card style={{overflow:"hidden"}}>
      <div style={{padding:"14px 16px",borderBottom:`1px solid ${T.border}`,display:"flex",gap:8,alignItems:"center",flexWrap:"wrap"}}>
        <span style={{fontWeight:800,fontSize:14,color:T.navy}}>📋 在庫管理追加版 — 納品ケース数</span>
        <input value={search} onChange={e=>setSearch(e.target.value)} placeholder="品名・JAN・メーカーで検索..."
          style={{flex:1,minWidth:160,padding:"6px 11px",border:`1px solid ${T.border}`,borderRadius:8,fontSize:12,outline:"none",fontFamily:font}}/>
        <select value={filterMonth||""} onChange={e=>setFilterMonth(e.target.value?parseInt(e.target.value):null)}
          style={{padding:"5px 9px",border:`1px solid ${T.border}`,borderRadius:8,fontSize:12,fontFamily:font,cursor:"pointer"}}>
          <option value="">全月</option>
          {MONTH_ORDER.map(m=><option key={m} value={m}>{m}月</option>)}
        </select>
        <span style={{fontSize:12,color:T.textLight}}>{entries.length}件</span>
      </div>

      <div style={{overflowX:"auto"}}>
        <table style={{width:"100%",borderCollapse:"collapse",fontSize:12}}>
          <thead>
            <tr style={{background:T.navyMid}}>
              <th style={{padding:"7px 12px",color:"#fff",fontWeight:700,textAlign:"left",whiteSpace:"nowrap",position:"sticky",left:0,background:T.navyMid,zIndex:2,minWidth:160}}>品名</th>
              <th style={{padding:"7px 10px",color:"#fff",fontWeight:700,textAlign:"left",whiteSpace:"nowrap",minWidth:100}}>メーカー</th>
              {MONTH_ORDER.map(m=>(
                <th key={m} style={{padding:"7px 7px",color:m===currentMonth?T.tealLight:"#fff",fontWeight:700,textAlign:"center",whiteSpace:"nowrap",background:m===currentMonth?T.tealDark:T.navyMid,minWidth:42}}>{m}月</th>
              ))}
            </tr>
          </thead>
          <tbody>
            {visible.map(([jan,item],i)=>{
              const hl=highlightJans.includes(jan);
              const bg=hl?"#CCFBF1":i%2===0?T.surfaceAlt:T.surface;
              return(
                <tr key={jan} style={{background:bg,borderBottom:`1px solid ${T.border}`}}>
                  <td style={{padding:"6px 12px",fontWeight:hl?800:600,color:T.navy,maxWidth:180,overflow:"hidden",textOverflow:"ellipsis",whiteSpace:"nowrap",position:"sticky",left:0,background:bg,zIndex:1,borderRight:`1px solid ${T.border}`}}>
                    {hl&&<span style={{color:T.teal,marginRight:3}}>★</span>}{item.name}
                  </td>
                  <td style={{padding:"6px 10px",color:T.textMid,whiteSpace:"nowrap",overflow:"hidden",maxWidth:100,textOverflow:"ellipsis"}}>{item.manufacturer}</td>
                  {MONTH_ORDER.map(m=>{
                    const v=item.deliveries[m]; const isCur=m===currentMonth;
                    return<td key={m} style={{padding:"6px 7px",textAlign:"center",fontWeight:v?700:400,color:v?(isCur?T.tealDark:T.teal):T.textLight,background:isCur&&v?"#CCFBF166":"transparent"}}>{v!=null&&v!==""?v:"—"}</td>;
                  })}
                </tr>
              );
            })}
          </tbody>
        </table>
        {!showAll&&entries.length>40&&<div style={{padding:"10px",textAlign:"center"}}><Btn onClick={()=>setShowAll(true)} bg={T.navyLight} size="sm">全{entries.length}件を表示</Btn></div>}
        {entries.length===0&&<div style={{padding:"28px",textAlign:"center",color:T.textLight,fontSize:14}}>該当なし</div>}
      </div>
    </Card>
  );
}

// ─── Main ─────────────────────────────────────────────────────────────────────
export default function App(){
  const [xlsxFile,setXlsxFile]=useState(null);
  const [inventoryMap,setInventoryMap]=useState(null);
  const [loadErr,setLoadErr]=useState(null);
  const [loading,setLoading]=useState(false);
  const [invoice,setInvoice]=useState(null);
  const [history,setHistory]=useState([]);
  const [recentJans,setRecentJans]=useState([]);
  const [toast,setToast]=useState(null);

  const currentMonth=todayMonth();
  const notify=(msg,type="ok")=>{setToast({msg,type});setTimeout(()=>setToast(null),4500);};

  const loadExcel=async file=>{
    setLoadErr(null);setLoading(true);
    try{
      const buf=await file.arrayBuffer();
      const rows=await readXlsx(buf);
      const map=buildMap(rows);
      setXlsxFile(file);setInventoryMap(map);
      notify(`✅ Excelを読み込みました（${map.size}品目）`);
    }catch(e){setLoadErr("読み込み失敗: "+e.message);}
    finally{setLoading(false);}
  };

  const handleApply=(checkedRows,month,writeMode)=>{
    if(!inventoryMap)return;
    const newMap=new Map(inventoryMap);
    const applied=[],failed=[];
    for(const r of checkedRows){
      const jan=r._jan; const item=newMap.get(jan);
      if(!item){failed.push(r);continue;}
      const prev=item.deliveries[month];
      const qty=Number(r.quantity)||0;
      const newVal=writeMode==="add"?(Number(prev)||0)+qty:qty;
      newMap.set(jan,{...item,deliveries:{...item.deliveries,[month]:newVal}});
      applied.push({jan,name:item.name,month,quantity:qty,newVal,previous:prev});
    }
    setInventoryMap(newMap);
    setHistory(p=>[...p,...applied]);
    setRecentJans(applied.map(a=>a.jan));
    setInvoice(null);
    notify(`✅ ${applied.length}件を${month}月に反映${failed.length?`　⚠️ 未登録 ${failed.length}件`:""}`);
  };

  const handleExport=()=>{
    if(!inventoryMap)return;
    const fname=xlsxFile?xlsxFile.name.replace(/\.xlsx$/i,"_更新.csv"):"在庫管理更新.csv";
    exportCSV(inventoryMap,fname);
    notify("📥 CSVをダウンロードしました");
  };

  const totalItems=inventoryMap?.size||0;
  const monthDeliveries=inventoryMap?[...inventoryMap.values()].filter(v=>Number(v.deliveries[currentMonth])>0).length:0;

  return(
    <div style={{minHeight:"100vh",background:T.bg,fontFamily:font}}>
      {toast&&<Toast msg={toast.msg} type={toast.type}/>}

      <header style={{background:T.navy,padding:"0 20px",boxShadow:"0 2px 12px rgba(0,0,0,0.3)"}}>
        <div style={{maxWidth:1020,margin:"0 auto",display:"flex",alignItems:"center",justifyContent:"space-between",height:62}}>
          <div style={{display:"flex",alignItems:"center",gap:12}}>
            <div style={{width:40,height:40,borderRadius:10,background:T.teal,display:"flex",alignItems:"center",justifyContent:"center",fontSize:20}}>💊</div>
            <div>
              <div style={{fontWeight:900,fontSize:17,color:"#fff",letterSpacing:"-0.4px"}}>薬剤在庫管理システム</div>
              <div style={{fontSize:11,color:"rgba(255,255,255,0.45)"}}>納品仕切書スキャン → 在庫自動反映</div>
            </div>
          </div>
          {xlsxFile&&(
            <div style={{display:"flex",gap:8,alignItems:"center"}}>
              <div style={{background:"rgba(255,255,255,0.1)",borderRadius:7,padding:"4px 12px",fontSize:11,color:"rgba(255,255,255,0.75)"}}>📊 {xlsxFile.name}</div>
              <Btn onClick={handleExport} bg={T.tealDark} size="sm">⬇ CSV出力</Btn>
            </div>
          )}
        </div>
      </header>

      <main style={{maxWidth:1020,margin:"0 auto",padding:"22px 14px"}}>
        {/* Step 1: Excel */}
        <Card style={{padding:22,marginBottom:20}}>
          <div style={{display:"flex",alignItems:"center",gap:12,marginBottom:14}}>
            <div style={{width:28,height:28,borderRadius:"50%",background:T.amber,color:"#fff",fontWeight:900,fontSize:14,display:"flex",alignItems:"center",justifyContent:"center",flexShrink:0}}>1</div>
            <div>
              <div style={{fontWeight:800,fontSize:15,color:T.navy}}>{xlsxFile?"✅ Excelファイル読み込み済み":"Excelファイルを読み込む"}</div>
              <div style={{fontSize:12,color:T.textMid}}>{xlsxFile?`${xlsxFile.name}　（${totalItems}品目）`:"monthly_delivery_inventory_template_2.xlsx を選択してください"}</div>
            </div>
          </div>

          {loadErr&&<div style={{background:T.redBg,border:`1px solid ${T.red}40`,borderRadius:8,padding:"8px 14px",fontSize:12,color:T.red,marginBottom:12}}>⚠️ {loadErr}</div>}

          {loading
            ?<div style={{padding:"16px",textAlign:"center",color:T.teal,fontWeight:700,fontSize:14}}>⏳ Excelを解析中...</div>
            :<DropZone onFile={loadExcel} accept=".xlsx,.xls" label={xlsxFile?"別のExcelを読み込む":"Excelファイルを選択 / ドラッグ&ドロップ"} sub=".xlsx ファイル対応" icon="📊" color={T.amber}/>
          }

          {xlsxFile&&!loading&&(
            <div style={{display:"flex",gap:12,marginTop:14,flexWrap:"wrap"}}>
              {[{label:"登録品目数",value:totalItems,color:T.teal},{label:`${currentMonth}月 納品あり`,value:monthDeliveries,color:T.green},{label:"今セッション反映",value:history.length,color:T.navyLight}].map(s=>(
                <div key={s.label} style={{flex:1,minWidth:100,background:T.surfaceAlt,borderRadius:10,padding:"12px 16px",borderTop:`3px solid ${s.color}`}}>
                  <div style={{fontSize:11,color:T.textMid}}>{s.label}</div>
                  <div style={{fontSize:22,fontWeight:900,color:s.color,marginTop:2}}>{s.value}</div>
                </div>
              ))}
            </div>
          )}
        </Card>

        {xlsxFile?(
          <>
            {/* Step 2: Scan */}
            <div style={{display:"flex",alignItems:"center",gap:8,marginBottom:10}}>
              <div style={{width:28,height:28,borderRadius:"50%",background:T.teal,color:"#fff",fontWeight:900,fontSize:14,display:"flex",alignItems:"center",justifyContent:"center"}}>2</div>
              <span style={{fontWeight:700,fontSize:14,color:T.navy}}>納品仕切書をスキャン</span>
            </div>
            <Scanner onResult={r=>setInvoice(r)}/>

            {invoice&&<InvoiceConfirm invoice={invoice} inventoryMap={inventoryMap} onApply={handleApply} onDiscard={()=>setInvoice(null)}/>}

            <HistoryPanel history={history}/>

            {/* Step 3: Table */}
            <div style={{display:"flex",alignItems:"center",gap:8,marginBottom:10}}>
              <div style={{width:28,height:28,borderRadius:"50%",background:T.navyLight,color:"#fff",fontWeight:900,fontSize:14,display:"flex",alignItems:"center",justifyContent:"center"}}>3</div>
              <span style={{fontWeight:700,fontSize:14,color:T.navy}}>在庫一覧を確認 → CSV出力</span>
              {recentJans.length>0&&<span style={{fontSize:12,color:T.textLight}}>（★ = 今回反映）</span>}
            </div>
            <InventoryTable inventoryMap={inventoryMap} highlightJans={recentJans} currentMonth={currentMonth}/>
          </>
        ):(
          <Card style={{padding:"44px 24px",textAlign:"center"}}>
            <div style={{fontSize:52,marginBottom:14}}>📊</div>
            <div style={{fontSize:17,fontWeight:800,color:T.navy,marginBottom:8}}>まずExcelファイルを読み込んでください</div>
            <div style={{fontSize:13,color:T.textMid}}>.xlsx ファイルをドラッグ&ドロップ、またはクリックして選択</div>
          </Card>
        )}
      </main>
    </div>
  );
}
